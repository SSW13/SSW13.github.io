DROP TABLE IF EXISTS `food`;
CREATE TABLE IF NOT EXISTS `food` (
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'รหัสสินค้า',
  `name` varchar(30) COLLATE utf8_bin NOT NULL COMMENT 'ชื่อสินค้า',
  `description` text COLLATE utf8_bin NOT NULL COMMENT 'คำอธิบาย',
  `price` double COLLATE utf8_bin NOT NULL COMMENT 'ราคา',
  `image` varchar(50) COLLATE utf8_bin NOT NULL COMMENT 'รูป',
  `cat_id` int(11) COLLATE utf8_bin NOT NULL COMMENT 'รหัสประเภท',
  PRIMARY KEY (`id`)
) ENGINE=MyISAM  DEFAULT CHARSET=utf8 COLLATE=utf8_bin AUTO_INCREMENT=1 ;
